;
; Binary Print Test
;
        NAME    "Binary Print"
        
        COPYI   S,0xFFFF
        COPYI   P,TESTW
        LOAD    P,P
        CALL    PBINS
        HALT

TESTW:  PUT     0xABCD;

;REGION Print Bits
PBINS:  COPYI   L,16    ; L = 16
PBINSL: DEC     L       ; L--
        COPY    P,M     ; M = P
        COPY    L,N     ; N = L
        SHR     M,N     ; N = M >> N
        ANDI    N,1     ; N = N & 1 
        ADDI    N,0x30  ; N = N + 0x30
        COPY    N,A     ; A = N
        OUTI    A,0x0A  ; A -> Out (0xA)
        REZ     L
        JUMP    PBINSL
;ENDREGION

        HALT
;       END
